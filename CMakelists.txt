cmake_minimum_required(VERSION 3.10)
project(FlexibleAISystem)

# إعدادات C++
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# إنشاء التنفيذي
add_executable(flexible_system main_system.cpp)

# إعدادات التصحيح
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    target_compile_definitions(flexible_system PRIVATE DEBUG_MODE)
endif()
